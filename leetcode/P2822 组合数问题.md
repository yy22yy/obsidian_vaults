
[[组合数学]]

对于含组合数的计算，可以考虑写出公式然后化简，最后通过某个自增量i的关系写出最后组合式的值

## 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
long long comb(int n, int r) {
    if (r > n) return 0;
    if (r > n - r) r = n - r;//r始终为较小者，利用对称性统一公式
    long long res = 1;
    int uptimes = 1;
    int downtimes = 1;
    for (int i = 0; i < r; i++){
            
        res = res * (n - i) / (i + 1);
    }
        //上面这个写法可能出现除法截断
        //i+1从(1,r);n-i从(n,n-r+1);
        //先分别累乘在除以起来
    return res;
}
int main(){
    int t,k;
    cin>>t>>k;
    
    for(int w= 0;w<t;w++){
        int n,m;cin>>n>>m;
        int count = 0;//本行的统计整数
        for(int i = 0;i<=n;i++){
            for(int j=0;j<=min(i,m);j++){
            long long res = comb(i,j);
            if(res%k == 0 && res>=k)  count ++;
                //cout<<i<<" "<<j<<" "<<res<<" "<<k<<endl;
            }
        }
        cout<< count <<endl;
        
    }
    return 0;
}
```
