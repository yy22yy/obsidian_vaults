#include <iostream>
#include <iomanip>
#include <vector>
using namespace std;

void backtrack(int n, vector<int>& path, vector<bool>& used) {
    // n是排列长度，path是已经构造的序列，used表示哪些数字已经使用过了
    if (path.size() == n) {
        // 输出当前排列
        for (int num : path) {
            cout << setw(5) << num;
        }
        cout << endl;
        return;
    }
    
    for (int i = 1; i <= n; i++) {
        if (!used[i]) {
            used[i] = true;           // 标记已使用
            path.push_back(i);        // 加入当前排列
            
            //最主要是这样一步，进入了一个状态
            backtrack(n, path, used); // 递归
            
            path.pop_back();          // 回溯
            used[i] = false;
        }
    }
}

int main() {
    int n;
    cin >> n;
    
    vector<int> path;      // 存储当前排列
    vector<bool> used(n + 1, false);  // 标记数字是否使用过
    
    backtrack(n, path, used);
    
    return 0;
}
