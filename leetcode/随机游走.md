
```cpp
#include <iostream>
#include <cstdlib>
#include <ctime>

struct Point {
    int x, y;
};

int main() {
    srand(time(0));
    
    Point pos = {0, 0}; // 初始坐标
    int steps = 20;
    
    // 四个方向：上下左右
    int dx[] = {0, 0, -1, 1};
    int dy[] = {1, -1, 0, 0};
    
    std::cout << "Start: (" << pos.x << ", " << pos.y << ")\n";
    
    for (int i = 0; i < steps; i++) {
        int dir = rand() % 4; // 随机选一个方向
        pos.x += dx[dir];
        pos.y += dy[dir];
        std::cout << "Step " << i+1 << ": (" << pos.x << ", " << pos.y << ")\n";
    }
    
    return 0;
}

```

steps：总共要走动的步数；

数组dx、dy：用于存储可能的方向；

dir = rand() % 4; // 随机选一个方向；
	如果dx与dy的数量不同，另外定义一个dir2对对应数组长度进行取余；
