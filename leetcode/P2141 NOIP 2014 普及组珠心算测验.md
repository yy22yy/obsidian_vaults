
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> arr(n);
    
    unordered_set<int> numSet;  // 存所有数，用于快速查找
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
        numSet.insert(arr[i]);
    }
    
    sort(arr.begin(), arr.end());
    
    int count = 0;
    unordered_set<int> found;  // 防止重复计数
    
    // 枚举所有两数之和
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            int sum = arr[i] + arr[j];
            // 和在集合中，且没被统计过
            if (numSet.count(sum) && !found.count(sum)) {
                count++;
                found.insert(sum);
            }
        }
    }
    
    cout << count << endl;
    return 0;
}
```
