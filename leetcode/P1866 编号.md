```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;cin>>n;

    const long long MOD = 1e9 + 7;//定义取余常量
    vector<int> arr(n+1,0);
    //完成对兔子喜爱编号的输入
    for(int i = 1;i<=n;i++){
        cin>>arr[i];
    }
    sort(arr.begin(),arr.end());
    long long int count =1;
    for(int i=1;i<=n;i++){
        count = count*(arr[i]-(i-1));
        if(count<=0) {cout<<0;return 0;}
        count = count % MOD;  取余需要提前放到此处
    }
    cout <<count;
    return 0;
}
```

- 对于数学问题时长要考虑表示类型的溢出问题
	>此题的取余不能放在for循环外面，要边算边取余
