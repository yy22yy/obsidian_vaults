
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n , m , k;
    cin >> n >> m >> k;
	
    vector<int> A(n*n, 0);
	
	//向量输入
    int x, y;

    // 🔥 火把
    
    /// 处理第一个3*3矩阵 step2：处理十字区域
    for(int i = 0; i < m; i++){
        cin >> x >> y;
        x--; y--;   // 0-based

        // 1️⃣ 先点亮 3×3
        for(int dx = -1; dx <= 1; dx++){
            for(int dy = -1; dy <= 1; dy++){
                int nx = x + dx;
                int ny = y + dy;
                if(nx>=0 && nx<n && ny>=0 && ny<n)
                    A[nx*n + ny] = 1;
            }
        }

        // 2️⃣ 再点亮 十字延伸
        int dx4[4] = {2,-2,0,0};
        int dy4[4] = {0,0,2,-2};

        for(int t = 0; t < 4; t++){
            int nx = x + dx4[t];
            int ny = y + dy4[t];
            if(nx>=0 && nx<n && ny>=0 && ny<n)
                A[nx*n + ny] = 1;
        }
    }

    // 荧光石
    for(int i = 0; i < k; i++){
        cin >> x >> y;
        x--; y--;

        for(int dx = -2; dx <= 2; dx++){
            for(int dy = -2; dy <= 2; dy++){
                int nx = x + dx;
                int ny = y + dy;
                if(nx>=0 && nx<n && ny>=0 && ny<n)
                    A[nx*n + ny] = 1;
            }
        }
    }

    int cnt = count(A.begin(), A.end(), 0);
    cout << cnt;

    return 0;
}
```
