# 无优化解

```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    //这个就是最短二叉树问题
    //哈夫曼树，优先合并小的
    int n;cin>>n;vector<int> arr(n);
    for(int i=0;i<n;i++){int x;cin>>x;arr[i]=x;}
    
    // 需要构造树吗？每次-2+1=-1，要选中n-1次
    // 需要构造一棵树，并且进行替代
    int sum =0 ;
    while(arr.size() >1){
        sort(arr.begin(),arr.end(),greater<int>());
        int x = arr.back();arr.pop_back();
        int y= arr.back();arr.pop_back();
        int sum1 = x+y;
        sum += x +y;
        arr.push_back(sum1);
    }
    cout << sum;
}
```

# 使用优先队列 -- [[priority_queue 堆的建立]]

- 底层存储：vector数组；
- 基本声明

```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n; cin >> n;
    // 小根堆：直接用 vector 初始化，O(n) 建堆
    priority_queue<long long, vector<long long>, greater<long long>> pq;
    for(int i = 0; i < n; i++){
        int x; cin >> x;
        pq.push(x);
    }

    long long sum = 0;
    while(pq.size() > 1){
        long long x = pq.top(); pq.pop();   // O(log n)
        long long y = pq.top(); pq.pop();   // O(log n)
        sum += x + y;
        pq.push(x + y);                     // O(log n)
    }
    cout << sum;
}
```
